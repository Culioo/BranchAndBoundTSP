37 void compute_minimal_1_tree(TSP::OneTree &tree,
38  const std::vector<double> &lambda,
39  const TSP::Instance<coord_type, dist_type> &tsp,
40  const TSP::BranchingNode<coord_type, dist_type> &BNode) {
41 
42  //compute modified weights c_\lambda and set the weight of required edges
43  // to -inf and for forbidden edges to +inf
44  std::vector<dist_type> mod_weights(tsp.num_edges(), 0);
45 
46  for (size_t edge = 0; edge < tsp.weights().size(); edge++) {
47  TSP::NodeId v = 0, w = 0;
48  to_NodeId(edge, v, w, tsp.size());
49  mod_weights.at(edge) = tsp.weight(edge) + lambda[v] + lambda[w];
50  }
51 
52  for (const auto &el : BNode.get_forbidden())
53  mod_weights.at(el) = std::numeric_limits<dist_type>::max();
54  for (const auto &el : BNode.get_required())
55  mod_weights.at(el) = -1; //std::numeric_limits<dist_type>::min();
56 
57 
58  // computing a MST on {2,..,n} by PRIM MST Algorithm
59  typedef std::pair<dist_type, int> Pair;
60  //instantiate our priority_queue properly
61  std::priority_queue<Pair, std::vector<Pair>, std::greater<Pair> > pq;
62 
63  int src = 1; // Start at the first node != 0
64  TSP::size_type n = tsp.size();
65  // First, make all nodes unreachable
66  std::vector<double> key(n, ::std::numeric_limits<double>::max() / 2.);
67 
68  // parent will give access to the second node in an edge for the MST
69  std::vector<int> parent(n, -1);
70 
71  // included vertices vector
72  std::vector<bool> MST_contained(n, false);
73  //start with the source....
74  pq.push(std::make_pair(0, src));
75  key[src] = 0;
76 
77  while (!pq.empty()) {
78  TSP::NodeId u = pq.top().second;
79  pq.pop();
80 
81  MST_contained[u] = true; // Include vertex in MST
82 
83  for (TSP::NodeId i = 1; i < n; i++) {
84  if (i != u) {
85  dist_type weight = mod_weights.at(to_EdgeId(u, i, n));
86  if (MST_contained[i] == false && key[i] > weight) {
87  // Updating key of i
88  key[i] = weight;
89  pq.push(std::make_pair(key[i], static_cast<int>(i)));
90  parent[i] = u;
91  }
92  }
93  }
94  }
95  //Done with MST computation, add them to our tree
96  for (TSP::NodeId k = 2; k < n; k++) {
97  tree.add_edge(k, static_cast<NodeId >(parent[k]));
98  }
99 
100  //seek for smallest two edges incident to 0 ..
101  TSP::NodeId smallest = 1;
102  for (TSP::NodeId k = 2; k < n; k++) {
103  if (mod_weights.at(to_EdgeId(0, k, n)) < mod_weights.at(to_EdgeId(0, smallest, n))) {
104  smallest = k;
105  }
106  }
107 
108  TSP::NodeId smallest1 = 1;
109  if (smallest == 1) smallest1 = 2;
110  for (TSP::NodeId k = 2; k < n; k++) {
111  if (k != smallest) {
112  if (mod_weights.at(to_EdgeId(0, k, n)) < mod_weights.at(to_EdgeId(0, smallest1, n))) {
113  smallest1 = k;
114  }
115  }
116  }
117  // ..add them
118  tree.add_edge(0, smallest);
119  tree.add_edge(0, smallest1);
120 }

134 template<class coord_type, class dist_type>
135 dist_type Held_Karp(const TSP::Instance<coord_type, dist_type> &tsp,
136  std::vector<double> &lambda,
137  TSP::OneTree &tree,
138  const TSP::BranchingNode<coord_type, dist_type> &bn,
139  bool root = false) {
140  // Initialization
141  TSP::size_type n = tsp.size();
142  std::vector<dist_type> sol_vector;
143  std::vector<double> lambda_max(lambda.size(), 0), lambda_tmp(lambda);
144  TSP::OneTree tree_max(tree), tree_tmp(tree);
145  double t_0 = 0., del_0 = 0., deldel = 0.;
146  size_t max_el = 0;
147  size_t N = std::ceil(n / 4.) + 5;
148  if (root) {
149  N = std::ceil(n * n / 50.) + n + 15;
150  }
151  // First tree computation to obtain t_0, del_0 , deldel
152  compute_minimal_1_tree<coord_type, dist_type>(tree, lambda_tmp, tsp, bn);
153  if (root) {
154  dist_type sum = 0;
155  for (const auto &el : tree.get_edges())
156  sum += tsp.weight(el);
157  t_0 = sum / (2. * n);
158  } else {
159  t_0 = 0;
160  for (TSP::NodeId i = 0; i < n; i++) {
161  t_0 += fabs(lambda.at(i));
162  }
163  t_0 *= 1. / (2. * n);
164  }
165 
166  deldel = t_0 / (N * N - N);
167  del_0 = 3. * t_0 / (2. * N);
168 
169  for (size_t i = 0; i < N; i++) {
170  //Computing the sum we later on want to maximize over
171  dist_type sum = 0, sum2 = 0;
172  for (const auto &el : tree.get_edges())
173  sum += tsp.weight(el);
174  for (size_t node = 0; node < n; node++)
175  sum2 += (tree.get_node(node).degree() - 2.) * lambda_tmp[node];
176  sol_vector.push_back(sum + sum2); //we save all, not necessary, but nice for understanding
177 
178  if (i == 0) { // the first iteration is slightly different..
179  tree_max = tree;
180  lambda_max = lambda_tmp;
181 
182  for (size_t j = 0; j < lambda_tmp.size(); j++) {
183  lambda_tmp[j] += t_0 * (tree.get_node(j).degree() - 2.);
184  }
185  t_0 = t_0 - del_0;
186  del_0 = del_0 - deldel;
187  tree_tmp = tree;
188  }
189 
190  if (i > 0) { // ..then this one
191  if (sol_vector[max_el] < sol_vector[i]) {
192  lambda_max = lambda_tmp;
193  max_el = i;
194  tree_max = tree;
195  }
196  for (size_t j = 0; j < lambda_tmp.size(); j++) {
197  lambda_tmp[j] +=
198  t_0 * (0.6 * (tree.get_node(j).degree() - 2.) + 0.4 * (tree_tmp.get_node(j).degree() - 2.));
199  }
200  t_0 = t_0 - del_0;
201  del_0 = del_0 - deldel;
202  tree_tmp = tree;
203  }
204  tree = TSP::OneTree(n);
205  compute_minimal_1_tree<coord_type, dist_type>(tree, lambda_tmp, tsp, bn);
206  }
207  if (root) { //Setting the holy lambda
208  lambda = lambda_max;
209  }
210  tree = tree_max;
211  // Multiplying by 1. - EPS whereas EPS is a Macro defined to 10e-7 since we do not want to
212  // obtain a lower bound larger than the optimum solution. This could occur due to
213  // floating point computations .
214  return std::ceil((1. - EPS) * (*std::max_element(sol_vector.begin(), sol_vector.end())));
215 }

275 void Instance<coord_type, dist_type>::compute_optimal_tour() {
276  typedef BranchingNode<coord_type, dist_type> BNode;
277 
278  dist_type upperBound = std::numeric_limits<dist_type>::max();
279  std::priority_queue<BNode,
280  std::vector<BNode>, std::greater<BNode> > Q;
281  Q.push(BranchingNode<coord_type, dist_type>(*this)); // Adding empty node to Q
282 
283  while (!Q.empty()) {
284  BNode current_BNode(Q.top());
285  Q.pop();
286  if (current_BNode.get_HK() >= upperBound)
287  continue;
288  else {
289  if (current_BNode.tworegular()) {
290  upperBound = current_BNode.get_HK();
291  std::cerr << "Upper Bound " << upperBound << std::endl;
292  _tour = current_BNode.get_tree().get_edges();
293  continue;
294  } else {
295  size_type gl_i = 0, choice1 = std::numeric_limits<size_type>::max(),
296  choice2 = std::numeric_limits<size_type>::max();
297  for (NodeId node = 1; node < current_BNode.get_tree().get_nodes().size(); node++) {
298  if (current_BNode.get_tree().get_node(node).degree() > 2) {
299  gl_i = node;
300  break;
301  }
302  }
303 
304  assert(gl_i != 0);
305  size_t counter = 0;
306  for (const auto &el : current_BNode.get_tree().get_node(gl_i).neighbors()) {
307  if (!current_BNode.is_required(to_EdgeId(gl_i, el, size()))) {
308  assert(!current_BNode.is_forbidden(to_EdgeId(gl_i, el, size())));
309  if (counter == 0)
310  choice1 = el;
311  if (counter == 1)
312  choice2 = el;
313  counter++;
314  if (counter > 1)
315  break;
316  }
317  }
318  assert(choice1 < std::numeric_limits<NodeId>::max());
319  assert(choice2 < std::numeric_limits<NodeId>::max());
320  BNode q1(current_BNode, *this, to_EdgeId(gl_i, choice1, this->size())),
321  q2(current_BNode,
322  *this,
323  to_EdgeId(gl_i, choice1, this->size()),
324  to_EdgeId(gl_i, choice2, this->size()));
325  Q.push(q1);
326  Q.push(q2);
327 
328  if (!current_BNode.get_required_neighbors().at(gl_i).degree()) {
329  BNode q3(current_BNode, *this,
330  to_EdgeId(gl_i, choice1, this->size()),
331  to_EdgeId(gl_i, choice2, this->size()),
332  true);
333  Q.push(q3);
334  }
335  }
336  }
337  }
338  std::cerr << "Optimal Length " << upperBound << std::endl;
339 
340  this->_length = upperBound;
341 }
